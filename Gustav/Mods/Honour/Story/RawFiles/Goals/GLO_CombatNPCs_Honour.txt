//When the Dummy spell goes off, create a target at every party member in the fight that is not hidden
IF
CastedSpell(_Titan, "Projectile_LOW_WatcherTitan_DummyHellfireMissiles", _, _, _StoryID)
AND
DB_HonorMode_Enabled(1)
AND
DB_Is_InCombat(_Titan, _CombatID)
AND
DB_PartyMembers(_Target)
AND
DB_Is_InCombat(_Target, _CombatID)
AND
NOT DB_HiddenCharacters((CHARACTER)_Target, _)
AND
CreateAtObject(Helper_Invisible_A_4cc75168-a81e-4a5c-85cd-1bab8d7bb641, _Target, 0, 0, "LOW_SteelWatchFoundery_TitanTarget_Created", 1, _TargetAOE)
THEN
SetCanJoinCombat(_TargetAOE, 0);
SetCanFight(_TargetAOE, 0);
ApplyStatus(_TargetAOE, "LOW_STEELWATCHFOUNDRY_TITAN_TARGETAREA", -1.0, 1, NULL_00000000-0000-0000-0000-000000000000);
DB_TestMissileID(_Titan, _TargetAOE);

QRY
QRY_LOW_SteelWatchFoundry_TitanHasPendingMissiles((CHARACTER)_Titan)
AND
DB_TestMissileID(_Titan, _)
THEN
DB_NOOP(1);


//When the turn starts and there are any targets, do a 1 second wait
IF
TurnStarted(_Titan)
AND
DB_HonorMode_Enabled(1)
AND
QRY_LOW_SteelWatchFoundry_TitanHasPendingMissiles((CHARACTER)_Titan) //checking if there are any missiles to shoot without iterating through the DB
AND
NOT QRY_Combat_TitanIncapacitated(_Titan)
THEN
SetEntityEventReal(_Titan,"GLO_CombatWait",1.0);
RealtimeObjectTimerLaunch(_Titan,"LOW_SteelWatchFoundry_TestMissiles_Honor",1000);

//When the 1 second wait is done create a disposable list of targets and then fire at each of them with a 0.5 second delay between shots
IF
ObjectTimerFinished(_Titan,"LOW_SteelWatchFoundry_TestMissiles_Honor")
AND
DB_TestMissileID(_Titan,_Target)
THEN
DB_SteelWatchFoundry_MissileTargets(_Target);

IF
ObjectTimerFinished(_Titan,"LOW_SteelWatchFoundry_TestMissiles_Honor")
THEN
UseSpell(_Titan, "ProjectileStrike_SteelWatcher_Quadruped_HellfireMissiles_AnimationOnly", _Titan);
SetEntityEventReal(_Titan,"GLO_CombatWait",0.5);
TimerLaunch("LOW_SteelWatchFoundry_Stagger",500);

PROC
PROC_SteelWatchFoundry_CastHellfireAtAllTargetsWithDelay()
AND
DB_SteelWatchFoundry_MissileTargets(_Target)
AND
DB_TestMissileID(_Titan, _Target)
AND
QRY_OnlyOnce("LOW_SteelWatchFoundry_HellfireMissilesPause")
THEN
UseSpell(_Titan, "ProjectileStrike_SteelWatcher_Quadruped_HellfireMissiles_DamageOnly", _Target);
NOT DB_SteelWatchFoundry_MissileTargets(_Target);

IF
TimerFinished("LOW_SteelWatchFoundry_Stagger")
AND
QRY_OnlyOnce_Reset("LOW_SteelWatchFoundry_HellfireMissilesPause")
THEN
PROC_SteelWatchFoundry_CastHellfireAtAllTargetsWithDelay();

THEN

SetEntityEventReal(_Titan,"GLO_CombatWait",0.5);
TimerLaunch("LOW_SteelWatchFoundry_Stagger",500);
